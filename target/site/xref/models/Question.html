3   import lombok.extern.slf4j.Slf4j;
4   import org.tinylog.Logger;
5   
6   import java.util.Random;
7   
8   /**
9    * The Question model is the base of the entire program. It consists of two Strings: a question and an answer.
10   * The class itself contains the getters-and-setters, and the {@code loadNewQuestion()} function. This instantiates
11   * an {@link PersonalXmlReader} with a random number passed to it as a parameter, and we pass that randomly loaded question
12   * on to the question, and return with the question itself.
13   */
14  
15  @Slf4j
16  public class Question {
17  
18      private String question;
19      private String answer;
20  
21      public String getQuestion() {
22          return question;
23      }
24  
25      public void setQuestion(String question) {
26          this.question = question;
27      }
28  
29      public String getAnswer() {
30          return answer;
31      }
32  
33      public void setAnswer(String answer) {
34          this.answer = answer;
35      }
36  
37      public static Question loadNewQuestion(){
38          Question question = new Question();
39          Random random = new Random();
40          int rand = random.nextInt(70);
41          Logger.info("Passing randomly generated Integer to the XML reader.");
42          PersonalXmlReader reader = new PersonalXmlReader();
43          String[] data = reader.XmlReader(rand);
44          question.setQuestion(data[0]);
45          question.setAnswer(data[1]);
46          return question;
47      }
48  }
