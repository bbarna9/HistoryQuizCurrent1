17  /**
18   * The {@code personalXmlReader} class executes the question loading. This is achieved by loading in the entire
19   * {@code questions.xml} file, and then selecting one specific question from the list, the one which's id equals
20   * with the number passed to the function as a parameter. After the question was selected, the function returns with it.
21   */
22  public class PersonalXmlReader {
23  
24      public String[] XmlReader(int temp){
25          String tempQ = "Error while loading question";
26          String tempA = "Error while loading answer";
27  
28          try {
29              DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
30              DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
31              Document doc = dBuilder.parse(getClass().getResourceAsStream("/xml/questions.xml"));
32              Logger.info("Reading XML file.");
33              doc.getDocumentElement().normalize();
34              NodeList nList = doc.getElementsByTagName("element");
35  
36              Node nNode = nList.item(temp);
37              if (nNode.getNodeType() == Node.ELEMENT_NODE) {
38                  Element eElement = (Element) nNode;
39                  tempQ = eElement.getElementsByTagName("text").item(0).getTextContent();
40                  tempA = eElement.getElementsByTagName("answer").item(0).getTextContent();
41                  Logger.info("Selecting random question");
42              }
43  
44          } catch (Exception e) {
45              e.printStackTrace();
46          }
47          String[] selectedQuestion = new String[] {tempQ, tempA};
48          return selectedQuestion;
49      }
50  }
